interface TripResponse {
  tripList: Trip[];
}

interface Trip {
  storeId: string;
  tripId: string;
  tripExternalId: string;
  batchId: string;
  tripStartDateTime: number;
  tripEndDateTime: number;
  totalDeliveryTasks: number;
  status: string;
  displayTripId: string;
  deliveryTasksByTenantId: {
    [tenantId: string]: DeliveryTask[];
  };
}

interface DeliveryTask {
  id: string;
  storeId: number;
  orderId: string;
  ringId: string;
  customerStartTime: number;
  customerEndTime: number;
  actualDeliveryTime: number;
  transitDeltaTime: number;
  taskEvents: TaskEvent[];
  externalStatus: string;
  serviceProvider: string;
  serviceType: string;
  deliveryStatus: string;
  carrier: string;
  transitStatus: string;
  dropoffEta: number;
  pickupEta: number;
  customerInfo: CustomerInfo;
  deliveryType: string;
  totalPlannedMileage: string;
  driver: Driver;
}

interface TaskEvent {
  eventStatus: string;
  eventTime: number;
}

interface CustomerInfo {
  customerFirstName: string;
  customerLastName: string;
  customerFullName: string;
  addressLine1: string;
  addressLine2: string;
  city: string;
  state: string;
  postalCode: string;
  country: string;
  primaryPhone: string;
}

interface Driver {
  driverUserId: string;
  firstName: string;
  lastName: string;
  phoneNumber: string;
}
